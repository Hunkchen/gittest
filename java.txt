#include<iostream>//顺序栈的创建
using namespace std;
#define STACK_INIT_SIZE 100
#define STACKINCREMENT 10
typedef int ElemType;
typedef struct SqStack{
	ElemType *base;
	ElemType *top;
	int stacksize;
}SqStack;
SqStack create_stack()
{
	SqStack S;
	S.base=new int[STACK_INIT_SIZE];
	S.top=S.base;
	S.stacksize=STACK_INIT_SIZE;
	return S;
}
void Push(SqStack&S,ElemType e)
{
	*S.top++=e;
}
void Pop(SqStack &S,ElemType&e)
{
	e=*--S.top;
}
void display(SqStack&S)
{
	while(S.top!=S.base){
		cout<<*(S.top-1)<<" ";
		S.top--;
	}
}
void create_bigger(SqStack&S)
{
	for(int i=0;i<STACKINCREMENT;i++)
	{
		cin>>S.base[i];
		S.top++;
	}
}
void main()
{
	void Push(SqStack&S,ElemType e);
	void create_bigger(SqStack&S);
	SqStack S;
	int e,m;
	S=create_stack();
	cout<<"create more data:"<<endl;
	create_bigger(S);
	cout<<"please enter the number e:"<<endl;
	cin>>e;
	Push(S,e);
	display(S);
	cout<<"delete number:"<<endl;
	Pop(S,m);
	display(S);
	cout<<m<<endl;
